/**
 * @param {Array<*>} xs
 * @param {Array<*>} ys
 * @param {function(*, *): *} f
 * @returns {Array<*>} ys
 * @test {zipWith([1, 2, 3], [3, 1, 0], (x, y) => x + y)} [4, 3, 3]
 */
const zipWith = (xs, ys, f) => {
  const end = Math.min(xs.length, ys.length);
  const zs = Array(end).fill(null);
  for (let i = 0; i < end; i++) {
    zs[i] = f(xs[i], ys[i]);
  }
  return zs;
};
